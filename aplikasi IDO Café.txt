        // Render menu items
        function renderMenuItems(items) {
            menuContainer.innerHTML = '';
            
            if (items.length === 0) {
                menuContainer.innerHTML = '<p class="empty-cart" style="grid-column: 1/-1; text-align: center; padding: 3rem;">Tidak ada menu yang ditemukan</p>';
                return;
            }
            
            items.forEach(item => {
                const isFavorite = favorites.includes(item.id);
                
                const menuCard = document.createElement('div');
                menuCard.className = 'menu-card';
                menuCard.dataset.category = item.category;
                menuCard.innerHTML = `
                    <div class="menu-image">
                        <i class="${item.icon}"></i>
                        <button class="favorite-btn ${isFavorite ? 'active' : ''}" data-id="${item.id}">
                            <i class="fas fa-heart"></i>
                        </button>
                    </div>
                    <div class="menu-content">
                        <span class="menu-category">${item.category === 'kopi' ? 'Kopi' : item.category === 'minuman' ? 'Minuman Non-Kopi' : item.category === 'makanan' ? 'Makanan Berat' : 'Snack'}</span>
                        <h3 class="menu-title">${item.name}</h3>
                        <div class="rating-stars">
                            ${renderStars(item.rating)}
                            <span>(${item.rating})</span>
                        </div>
                        <p class="menu-description">${item.description}</p>
                        <p class="menu-price">Rp ${item.price.toLocaleString()}</p>
                        <button class="order-btn" data-id="${item.id}" data-name="${item.name}" data-price="${item.price}">Tambah ke Pesanan</button>
                    </div>
                `;
                menuContainer.appendChild(menuCard);
                
                // Add click event to menu card
                menuCard.addEventListener('click', (e) => {
                    if (!e.target.closest('.favorite-btn') && !e.target.closest('.order-btn')) {
                        showProductDetail(item);
                    }
                });
            });
            
            // Add event listeners to favorite buttons
            document.querySelectorAll('.favorite-btn').forEach(button => {
                button.addEventListener('click', function(e) {
                    e.stopPropagation();
                    const id = parseInt(this.dataset.id);
                    toggleFavorite(id);
                });
            });
            
            // Add event listeners to order buttons
            document.querySelectorAll('.order-btn').forEach(button => {
                button.addEventListener('click', function(e) {
                    e.stopPropagation();
                    const id = parseInt(this.dataset.id);
                    const name = this.dataset.name;
                    const price = parseInt(this.dataset.price);
                    addToCart(id, name, price);
                });
            });

        // Render stars for rating
        function renderStars(rating) {
            let stars = '';
            const fullStars = Math.floor(rating);
            const halfStar = rating % 1 >= 0.5;
            
            for (let i = 1; i <= 5; i++) {
                if (i <= fullStars) {
                    stars += '<i class="fas fa-star"></i>';
                } else if (i === fullStars + 1 && halfStar) {
                    stars += '<i class="fas fa-star-half-alt"></i>';
                } else {
                    stars += '<i class="far fa-star"></i>';
                }
            }
            
            return stars;
        }

        // Toggle favorite
        function toggleFavorite(id) {
            if (!currentUser) {
                showNotification("Silahkan login untuk menambahkan favorit!");
                authModal.style.display = 'flex';
                return;
            }
            
            const index = favorites.indexOf(id);
            
            if (index === -1) {
                favorites.push(id);
                showNotification(`Ditambahkan ke favorit!`);
            } else {
                favorites.splice(index, 1);
                showNotification(`Dihapus dari favorit!`);
            }
            
            saveFavoritesToLocalStorage();
            updateFavoriteUI();
            renderMenuItems(menuItems);
            
            // If in favorite section, update the view
            if (favoriteSection.style.display === 'block') {
                renderFavoriteItems();
            }
        }
        
        // Add to cart
        function addToCart(id, name, price) {
            const existingItem = cart.find(item => item.id === id);
            
            if (existingItem) {
                existingItem.quantity++;
            } else {
                cart.push({
                    id,
                    name,
                    price,
                    quantity: 1
                });
            }
            
            updateCartUI();
            saveCartToLocalStorage();
            showNotification(`${name} ditambahkan ke keranjang!`);
        }
        
        // Update cart UI
        function updateCartUI() {
            // Update cart count
            const totalItems = cart.reduce((total, item) => total + item.quantity, 0);
            cartCount.textContent = totalItems;
            
            // Update cart items display
            cartItemsContainer.innerHTML = '';
            
            if (cart.length === 0) {
                cartItemsContainer.innerHTML = '<p class="empty-cart">Keranjang Anda kosong</p>';
                cartTotalElement.textContent = '0';
                return;
            }
            
            cart.forEach(item => {
                const cartItemElement = document.createElement('div');
                cartItemElement.className = 'cart-item';
                cartItemElement.innerHTML = `
                    <div class="cart-item-details">
                        <div class="cart-item-name">${item.name}</div>
                        <div class="cart-item-price">Rp ${item.price.toLocaleString()}</div>
                    </div>
                    <div class="cart-quantity">
                        <button class="quantity-btn minus" data-id="${item.id}">-</button>
                        <div class="quantity-display">${item.quantity}</div>
                        <button class="quantity-btn plus" data-id="${item.id}">+</button>
                    </div>
                `;
                cartItemsContainer.appendChild(cartItemElement);
            });
            
            // Update total
            const total = cart.reduce((total, item) => total + (item.price * item.quantity), 0);
            cartTotalElement.textContent = total.toLocaleString();
            
            // Add event listeners to quantity buttons
            document.querySelectorAll('.quantity-btn.minus').forEach(btn => {
                btn.addEventListener('click', function() {
                    const id = parseInt(this.dataset.id);
                    decreaseQuantity(id);
                });
            });
            
            document.querySelectorAll('.quantity-btn.plus').forEach(btn => {
                btn.addEventListener('click', function() {
                    const id = parseInt(this.dataset.id);
                    increaseQuantity(id);
                });
            });
        }

        // Increase item quantity
        function increaseQuantity(id) {
            const item = cart.find(item => item.id === id);
            if (item) {
                item.quantity++;
                updateCartUI();
                saveCartToLocalStorage();
            }
        }
        
        // Decrease item quantity
        function decreaseQuantity(id) {
            const itemIndex = cart.findIndex(item => item.id === id);
            if (itemIndex !== -1) {
                const item = cart[itemIndex];
                if (item.quantity > 1) {
                    item.quantity--;
                } else {
                    cart.splice(itemIndex, 1);
                }
                updateCartUI();
                saveCartToLocalStorage();
            }
        }

        // Filter menu items
        function filterMenuItems(category) {
            if (category === 'all') {
                renderMenuItems(menuItems);
                return;
            }
            
            const filteredItems = menuItems.filter(item => item.category === category);
            renderMenuItems(filteredItems);
        }
        
        // Search menu items
        function searchMenuItems(query) {
            if (!query) {
                renderMenuItems(menuItems);
                return;
            }
            
            const lowerQuery = query.toLowerCase();
            const filteredItems = menuItems.filter(item => 
                item.name.toLowerCase().includes(lowerQuery) || 
                item.description.toLowerCase().includes(lowerQuery)
            );
            
            renderMenuItems(filteredItems);
        }

            document.body.appendChild(notification);

            // Hide and remove after 3 seconds

                notification.classList.remove('show');
                setTimeout(() => {
                    notification.remove();

            });
            
            // Smooth scrolling for navigation
            document.querySelectorAll('nav a').forEach(anchor => {
                anchor.addEventListener('click', function(e) {
                    e.preventDefault();
                    const targetId = this.getAttribute('href');
                    
                    // If it's a section link
                    if (targetId.startsWith('#')) {
                        const targetElement = document.querySelector(targetId);
                        window.scrollTo({
                            top: targetElement.offsetTop - 80,
                            behavior: 'smooth'
                        });
                        
                        // Hide cart and favorite sections
                        cartSection.style.display = 'none';
                        favoriteSection.style.display = 'none';
                        profileSection.style.display = 'none';
                    }
                });
            });
            
            // Modal close events
            closeModal.addEventListener('click', closeProductModal);
            closeModalBtn.addEventListener('click', closeProductModal);
            closeAuthModal.addEventListener('click', function() {
                authModal.style.display = 'none';
            });
            
            // Modal tab switching
            document.querySelectorAll('.modal-tab').forEach(tab => {
                tab.addEventListener('click', function() {
                    const tabId = this.dataset.tab;
                    
                    document.querySelectorAll('.modal-tab').forEach(t => t.classList.remove('active'));
                    document.querySelectorAll('.tab-content').forEach(c => c.classList.remove('active'));
                    
                    this.classList.add('active');
                    document.getElementById(`${tabId}Content`).classList.add('active');
                });
            });
            
            // Add to cart from modal
            addToCartModal.addEventListener('click', function() {
                if (currentProduct) {
                    addToCart(currentProduct.id, currentProduct.name, currentProduct.price);
                    closeProductModal();
                }
            });
            
            // Close modal when clicking outside
            window.addEventListener('click', function(e) {
                if (e.target === productModal) {
                    closeProductModal();
                }
                if (e.target === authModal) {
                    authModal.style.display = 'none';
                }
            });
            
            // Auth tab switching
            authTabs.forEach(tab => {
                tab.addEventListener('click', function() {
                    const tabId = this.dataset.tab;
                    
                    authTabs.forEach(t => t.classList.remove('active'));
                    this.classList.add('active');
                    
                    if (tabId === 'login') {
                        loginContent.style.display = 'block';
                        registerContent.style.display = 'none';
                    } else {
                        loginContent.style.display = 'none';
                        registerContent.style.display = 'block';
                    }
                });
            });
            
            // Login button
            loginBtn.addEventListener('click', function() {
                const email = document.getElementById('loginEmail').value;
                const password = document.getElementById('loginPassword').value;
                login(email, password);
            });
            
            // Register button
            registerBtn.addEventListener('click', function() {
                const name = document.getElementById('registerName').value;
                const email = document.getElementById('registerEmail').value;
                const phone = document.getElementById('registerPhone').value;
                const password = document.getElementById('registerPassword').value;
                const confirmPassword = document.getElementById('registerConfirmPassword').value;
                
                if (password !== confirmPassword) {
                    showNotification('Password dan konfirmasi password tidak sama!');
                    return;
                }
                
                register(name, email, phone, password);
            });
            
            // Profile tab switching
            profileTabs.forEach(tab => {
                tab.addEventListener('click', function() {
                    const tabId = this.dataset.tab;
                    
                    profileTabs.forEach(t => t.classList.remove('active'));
                    profileContents.forEach(c => c.classList.remove('active'));
                    
                    this.classList.add('active');
                    document.getElementById(`${tabId}Content`).classList.add('active');
                    
                    // Populate form if settings tab
                    if (tabId === 'settings' && currentUser) {
                        document.getElementById('editName').value = currentUser.name;
                        document.getElementById('editEmail').value = currentUser.email;
                        document.getElementById('editPhone').value = currentUser.phone;
                        document.getElementById('editAddress').value = currentUser.address || '';
                    }
                });
            });
        }
